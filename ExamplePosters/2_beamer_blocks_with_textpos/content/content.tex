  % lets build our own environment with 
  % a minibox inside to be able to influence

  \newenvironment{mybox}[3]{
	  \begin{textblock*}{#1}#2
  	  \begin{minipage}[t][#3][t]{\textwidth}	
  }
  {
	    \end{minipage}
	  \end{textblock*}
